SELECT TOP 10 (EMPLOYEE_ID) FROM DLITHE.DBO.S_HREMPLOYEES

SELECT UPPER(FIRST_NAME) FROM DLITHE.DBO.S_HREMPLOYEES

SELECT LOWER(FIRST_NAME) FROM DLITHE.DBO.S_HREMPLOYEES

SELECT CONCAT(FIRST_NAME, ' ', LAST_NAME)FROM DLITHE.DBO.S_HREMPLOYEES

SELECT LTRIM(RTRIM('  NEHA LATHA   ')) FROM DLITHE.DBO.S_HREMPLOYEES

SELECT LEN ('NEHA LATHA') FROM DLITHE.DBO.S_HREMPLOYEES

SELECT REPLACE('NEHA LATHA',' ', '_')  AS FIRST_NAME  FROM DLITHE.DBO.S_HREMPLOYEES

SELECT CASE WHEN EMPLOYEE_ID =101 THEN 'MOODBIDRI'
WHEN EMPLOYEE_ID =102 THEN 'MANGALORE'
WHEN EMPLOYEE_ID =103 THEN 'KARKALA'
WHEN EMPLOYEE_ID =104 THEN 'UDUPI'
ELSE 'OTHERS'
END AS ADDRESS,EMPLOYEE_ID 
FROM DLITHE.DBO.S_HREMPLOYEES

SELECT SUBSTRING('NEHA LATHA',1,5) AS FIRST_NAME  FROM DLITHE.DBO.S_HREMPLOYEES

SELECT CHARINDEX ('T', 'NEHA LATHA') AS FIRST_NAME  FROM DLITHE.DBO.S_HREMPLOYEES
